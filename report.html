from sklearn.model_selection import train_test_split
from sklearn.tree import DecisionTreeRegressor

def get_all_features_score(features_list):
    msg_list = []
    for feature in features_list:
        new_data = data.drop([feature], axis = 1)
        feature_value = data[feature]

        # 
        X_train, X_test, y_train, y_test = train_test_split(new_data, feature_value, test_size=0.25, random_state=233)

        # DecisionTreeRegressor
        regressor = DecisionTreeRegressor(random_state=233)
        regressor.fit(X_train, y_train)

        # 
        score = regressor.score(X_test, y_test)
        
        # 
        msg_list.append('%s %.3f' % (feature, score))
    
    print('\n'.join(msg_list))
